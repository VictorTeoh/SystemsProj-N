echo -----
echo TESTING LS
echo -----
echo [ls]
ls
echo

echo -----
echo TESTING SEMICOLON
echo -----
echo [ls semicolon ps]
ls ; ps
echo

echo -----
echo TESTING REDIRECTION
echo -----
echo [ps (stdout redirect) test]
ps > test
echo [ls -a -l (stdout append) test]
ls -a -l >> test
echo [test (stdin redirect) cat]
cat < test
echo [rm test]
rm test
echo

echo -----
echo PIPING TEST
echo -----
echo [ls -a pipe wc]
ls -a | wc
echo

echo -----
echo CD TEST
echo -----
echo [cd ..]
cd ..
echo [cwd]
cwd
echo EXIT
exit
